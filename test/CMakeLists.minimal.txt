cmake_minimum_required(VERSION 3.10)
project(MinimalTextBufferTest)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add executable
add_executable(MinimalTextBufferTest
    test_minimal_textbuffer.cpp
    MinimalTextBuffer.h
)

# Enable testing
enable_testing()
add_test(NAME MinimalTextBufferTest COMMAND MinimalTextBufferTest)
